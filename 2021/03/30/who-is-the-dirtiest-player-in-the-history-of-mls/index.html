MLS <- read.csv("MLSDP.csv")
MLS %>% 
  group_by(Player) %>% 
  summarise(
    Reds = sum(CrdR),
    Yellows = sum(CrdY),
    Booking = sum(Reds, Yellows)
  ) %>%
arrange(desc(Booking)) %>% 
  top_n(10)-> MLSSB
## Selecting by Booking
MLSSBLong <- MLSSB %>% 
  pivot_longer(
    cols=("Yellows" | "Reds"), 
    names_to="Bookings", 
    values_to="Cards") %>%
  arrange(desc(Bookings)) %>%
  mutate(BookingOrder = factor(Bookings))
ggplot() + 
  geom_bar(data = MLSSBLong, aes(x=reorder(Player, Cards), weight=Cards, fill=BookingOrder)) +
  coord_flip() + 
  scale_fill_manual(values = c("red", "gold")) + 
  theme_minimal() +  
  labs(x="", y="Total Bookings", title="With 21 Seasons, Kyle Beckerman Has A Clear Lead In Bookings", subtitle="Other players however, have amassed more red cards than Beckerman in a lesser time period.", caption="Source: Sports Reference | By Peyton Thomas")  +
  theme(
    plot.title = element_text(size = 13, face = "bold"),
    axis.title = element_text(size = 10), 
    plot.subtitle = element_text(size=10), 
    panel.grid.minor = element_blank()
    )

FieldPlayers <- c("DF", "DFMF", "FW", "FWDF", "FWMF", "FWGK", "MF", "MFDF", "MFFW")
MLS %>% 
  group_by(Player) %>% 
  filter(
    X90s > 0,
    Pos %in% FieldPlayers) %>%
  summarize(
    Games = sum(`X90s`),
    YellowCards = sum(CrdY)) -> MLSAPP
MLSAPP %>% summarize(
  mean(Games),
  mean(YellowCards))

MLS %>% group_by(Player) %>% 
  filter(
    X90s > 0,
    Pos %in% FieldPlayers) %>%
  summarize(
    Bookings = sum(CrdY + CrdR),
    Years = n(),
    Games = sum(X90s),
    BPG = mean(Bookings/Games)) -> MLSFP
MLSFP %>% filter(
  Years > 5,
  Games > 100,
  BPG > .28858
) -> MLSFPD
MLSFP %>% filter(Player == "Kyle Beckerman") -> KB
ggplot() + 
  geom_point(
    data=MLSFP, 
    aes(x=Games, y=Bookings, size=Years), 
    color="grey", 
    alpha=.5) + 
  geom_point(
    data=MLSFPD, 
    aes(x=Games, y=Bookings, size=Years),

    alpha=.8) + 
  geom_point(
    data=KB, 
    aes(x=Games, y=Bookings, size=Years),

  ) + geom_smooth(data=MLSFP, aes(x=Games, y=Bookings), method="lm") + theme_minimal() +

Notorious <- MLS %>% group_by(Player) %>%
  summarize(
    Bookings = sum(CrdY + CrdR),
    Years = n(),
    Games = sum(X90s),
    YPG = mean(CrdY/Games),
    RPG = mean(CrdR/Games),
    FPG = mean(Fls/Games)) %>%
  filter(Games > 68,
         Bookings > 10,
         Years > 2) %>%
  ungroup()
Notorious %>% 
  mutate(Bonus = case_when(
  grepl("Diego C", Player) ~ 1, 
  grepl("", Player) ~ 0)) -> NameBonus
Dirtyness <- NameBonus %>% group_by(Player) %>%
  mutate(DirtynessRating = (((YPG*.34 + RPG*1.7 + FPG*.017)*(Games^.7))*5+(Bonus))) %>%
  ungroup()
Top10 <- Dirtyness %>%
 arrange(desc(DirtynessRating)) %>% 
  top_n(10, wt=DirtynessRating) 
Top10 %>% 
  select(Player, Years, Games, DirtynessRating) %>% 
  rename(`90s` = Games) %>% 
  gt() %>%
  tab_header(
    title = "Behind An Impressive Brigade Of Red Cards, Ashley Cole Tops The Pack",
    subtitle = "Diego Chara of the Portland Timbers also cracks the Top 5 in dirtiest players in history of the league, using an absolutely, 100% objective, non-bias statistic model."
  ) %>% tab_style(
